Forecasting Weekly Retail Demand at Walmart

Darian Miranda Olivas

Overview

End to end machine learning workflow for forecasting weekly Walmart retail sales at the store-department level using historical sales data.

Model Performance

Final model achieved 8.4% WAPE on the test dataset, demonstrating strong predictive performance on unseen future demand.

Data Source

This analysis uses the Walmart Recruiting – Store Sales Forecasting dataset made available through a Kaggle competition hosted by Walmart Inc. Access the data here.

#====== Create calendar features ======

calendar_fts_tbl <- base_tbl %>%
  distinct(week_dates) %>%
  mutate(
    
    #---- basic calendar features ----
    week    = week(week_dates),
    month   = month(week_dates),
    quarter = quarter(week_dates),
    year    = year(week_dates),
    
    #---- week index features ----
    week_of_month    = ceiling(day(week_dates) / 7), 
    week_of_quarter  = ceiling(yday(week_dates) / 7) - ceiling(yday(floor_date(week_dates, "quarter")) / 7), 
    
    #---- quarter edge features ----
    is_quarter_start = as.integer(week_of_quarter <= 2), ##quarter start defined as first 2 weeks
    is_quarter_end   = as.integer(week_of_quarter >= 12), ##quarter end defined as last 2 weeks
    
    #---- continuous seasonality features ----
    week_sin = sin(2 * pi * week / 52), ##sin and cos functions replicate cyclical nature of time
    week_cos = cos(2 * pi * week / 52)
    
  )

#====== Create a tbl with a feature for each holiday/event ======

holiday_event_tbl <- days_tbl %>% 
  mutate(
    
    #---- fixed date holiday features ----
    is_valentines     = month(day_dates) == 2  & mday(day_dates) == 14, 
    is_independence_day  = month(day_dates) == 7  & mday(day_dates) == 4,
    is_halloween      = month(day_dates) == 10 & mday(day_dates) == 31,
    is_christmas      = month(day_dates) == 12 & mday(day_dates) == 25,
    is_new_years_eve  = month(day_dates) == 12 & mday(day_dates) == 31,
    
    #---- pattern based holiday features ----
    is_super_bowl = ##1st sun of feb
      wday(day_dates) == 1 & 
      month(day_dates) == 2 & 
      mday(day_dates) <= 7,
   
    is_mothers_day    = #2nd sun of may
      wday(day_dates) == 1 &
      month(day_dates) == 5 &
      mday(day_dates) > 7 &
      mday(day_dates) <= 14,
      
     is_memorial_day   = ##last mon of may
      wday(day_dates) == 2 & 
      month(day_dates) == 5 & 
      mday(day_dates) >= 25,

    is_fathers_day    = #3rd sun of jun
      wday(day_dates) == 1 & 
      month(day_dates) == 6 & 
      mday(day_dates) > 14 & 
      mday(day_dates) <= 21,

    is_labor_day      = ##1st mon of sep
      wday(day_dates) == 2 & 
      month(day_dates) == 9 & 
      mday(day_dates) <= 7,
      
    is_thanksgiving = ##4th thu of nov 
      wday(day_dates) == 5 & 
      month(day_dates) == 11 & 
      mday(day_dates) >= 22 & 
      mday(day_dates) <= 28,

    is_black_friday = ##fri after thanksgiving
      wday(day_dates) == 6 &
      month(day_dates) == 11 &
      mday(day_dates) >= 23 &
      mday(day_dates) <= 29,
    
    is_easter = day_dates %in% easter_dates,
   
   #---- back to school ramp feature ----
    back_to_school_ramp = case_when( ##back to school happens at different times across the United States
      day_dates %in% 213:220 ~ 1, ##early back to school
      day_dates %in% 227:241 ~ 2, ##peak back to school
      day_dates %in% 248:255 ~ 1, ##late back to school
      TRUE ~ 0)

   )





##wday() function from Lubridate package acts inconsistently inside of mutate() function
##for a stable solution I define a weekday indexing function

#====== Define a weekday indexing function ======

wday_function <- function(day_dates) { 
  (wday(day_dates)) %% 7 + 1 ##first entry of data is 2010-02-05, so index Friday as 7
}





#====== Map holiday daily data to respective sales weeks ======

holiday_event_fts_tbl <- holiday_event_tbl %>%
  ##use algorithmic approach to round up to Friday because time round functions default to Sunday
  mutate(week_dates = day_dates +  7 - wday_function(day_dates)) %>% 
  group_by(week_dates) %>% 
  summarise(
    
    across(starts_with("is_"),
           ~ any(.x, na.rm = TRUE)), ##inserts TRUE for a holiday feature if any day of that sales week is a holiday
    
    back_to_school_ramp = max(back_to_school_ramp, na.rm = TRUE),
    
    .groups = "drop"
    
    )

lead_holiday_fts_tbl <- holiday_event_fts_tbl %>%
  select(week_dates, starts_with("is")) %>%
  arrange(week_dates) %>% 
  mutate(
    
     #---- 1 week lead for all holidays ----
     across(starts_with("is"),
           ~ lead(.x, 1), ##recall data is weekly so 1 period lead
           .names = "lead_{.col}_1"),
     
     #---- 2-3 weeks for major holidays only ----
     across(starts_with("is") & contains(major_holidays),
            list(
              `2` = ~ lead(.x, 2),
              `3` = ~ lead(.x, 3)
            ),
            .names = "lead_{.col}_{.fn}"),
     
     #---- 4 week lead for Christmas only ----
     across(is_christmas,
            ~ lead(.x, 4),
            .names = "lead_{.col}_4")
    
    ) %>%
    select(week_dates, contains("lead_is_"))

Demand Level And Variability

Purpose: To capture level and variability of retail demand in recent history.
Features Created
roll_mean_{4,8,13,26}

#====== Create demand level features ======

level_demand_fts_tbl <- lag_demand_fts_tbl %>%
  select(store, dept, week_dates, lag_weekly_sales_1) %>%
  arrange(store, dept, week_dates) %>%
  group_by(store, dept) %>%
  mutate(
    
    roll_mean_4  = slide_dbl(lag_weekly_sales_1, mean, .before = 3, .complete = TRUE), #.before = 3 means use current period and 3 before, gives 4 week window
    roll_mean_8  = slide_dbl(lag_weekly_sales_1, mean, .before = 7, .complete = TRUE),
    roll_mean_13 = slide_dbl(lag_weekly_sales_1, mean, .before = 12, .complete = TRUE),
    roll_mean_26 = slide_dbl(lag_weekly_sales_1, mean, .before = 25, .complete = TRUE),
    
    ) %>%
  ungroup() %>%
  select(store, dept, week_dates, contains("roll_mean_"))





#====== Create demand variability features ======

variability_demand_fts_tbl <- lag_demand_fts_tbl %>%
  select(store, dept, week_dates, lag_weekly_sales_1) %>%
  left_join(level_demand_fts_tbl %>%
      select(store, dept, week_dates, roll_mean_13),
      by = c("store", "dept", "week_dates")
      )%>%
  arrange(store, dept, week_dates) %>%
  group_by(store, dept) %>%
  mutate(
    
    roll_sd_4  = slide_dbl(lag_weekly_sales_1, sd, .before = 3, .complete = TRUE),
    roll_sd_8  = slide_dbl(lag_weekly_sales_1, sd, .before = 7,  .complete = TRUE),
    roll_sd_13 = slide_dbl(lag_weekly_sales_1, sd, .before = 12, .complete = TRUE),
    roll_sd_26 = slide_dbl(lag_weekly_sales_1, sd, .before = 25, .complete = TRUE),
    
    #---- demand shock feature ----
    is_demand_shock = abs(lag_weekly_sales_1 - roll_mean_13) > 2 * roll_sd_13 ##2 sd comes from statistical anomaly rule
    
    ) %>%
  ungroup() %>%
  select(store, dept, week_dates, contains(c("roll_sd_", "is")))

#====== Define a function that calculates slope ======

slope_function <- function(x) {
  
  time_index <- seq_along(x)
  lm_fit <- lm(x ~ time_index) ##regress sales on time
  
  coef(lm_fit)[2]
  
        }





#====== Define a function that calculates exponential weighted moving average (EWMA) ======

ewma_function <- function(x, alpha) {
  ewma_value <- x[1]
  
  for (i in 2:length(x)) {
    ewma_value <- alpha * x[i] + (1 - alpha) * ewma_value ##EWMA basic formula
  }
  
  ewma_value
}

    

#====== Create demand momentum features ======

momentum_demand_fts_tbl <- base_tbl %>%
  select(store, dept, week_dates, weekly_sales) %>%
   left_join(lag_demand_fts_tbl %>%
               select(store, dept, week_dates, lag_weekly_sales_1),
             by = c("store", "dept", "week_dates")
             ) %>%
  left_join(level_demand_fts_tbl %>%
              select(store, dept, week_dates, roll_mean_4, roll_mean_13),
            by = c("store", "dept", "week_dates")
            ) %>%
  arrange(store, dept, week_dates) %>% 
  group_by(store, dept) %>%
  mutate(
    
    #---- year over year growth feature ----
    growth_yoy = (lag_weekly_sales_1 - lag(weekly_sales, 53)) / pmax(lag(weekly_sales, 53), 1), ##prevent division by 0 with pmax(x,1)
    
    
    
    #---- short term vs medium term demand ratio feature ----
    ratio_demand_short_medium = roll_mean_4 / pmax(roll_mean_13, 1),
    
    
    
    #---- same week last year average feature ----
    avg_same_week_last_year = slide_dbl(
      lag(weekly_sales, 52), ##1 year prior
      mean,
      .before = 3, ##use a 4 week window since singular week sales can be noisy
      .complete = TRUE
      ),
    
    
    
    #---- rolling slope feature ----
    roll_slope_8 = slide_dbl(
      .x      = lag_weekly_sales_1,
      .before = 7,
      .f      = ~ slope_function(.x),
      .complete = TRUE
      ),
    
    
    
    #---- EWMA feature ----
    ewma_6 = slide_dbl( 
      .x      = lag_weekly_sales_1,             
      .before = 5,  
      .f      = ~ ewma_function(.x, alpha = 0.3),
      .complete = TRUE
      ),
    
    ) %>%
  ungroup() %>%
  select(store, dept, week_dates, contains(c("growth", "ratio", "avg", "roll_slope_", "ewma_")))

#====== Create mark down features ======

mark_down_fts_tbl <- base_tbl %>%
  select(store, dept, week_dates, starts_with("mark_down")) %>%
  left_join(lag_demand_fts_tbl %>%
              select(store, dept, week_dates, lag_weekly_sales_1),
            by = c("store", "dept", "week_dates")
            ) %>%
  mutate(
    
##NA in this data at times means no mark down and at other times means missing data
##to differentiate the two and extract signals I create NA features

    #---- mark down NA features ----
    across(starts_with("mark_down"),
           ~ is.na(.x),
           .names = "is_{.col}_na"),
    
    #---- total mark downs in a store department feature ----
    total_mark_down = rowSums(
      across(starts_with("mark_down")),
      na.rm = TRUE),
    
    #---- total mark down relative to store department demand feature ----
    total_intensity_mark_down = total_mark_down / pmax(lag_weekly_sales_1, 1),

    #---- individual mark downs relative to store department demand features ----
    across(starts_with("mark_down"),
      ~ .x / pmax(lag_weekly_sales_1, 1),
      .names = "intensity_{.col}")
    
    ) %>%
    select(store, dept, week_dates, contains(c("is", "total", "intensity")))

#====== Define a function to map splits to a data set ======

split_function <- function(split, data) {
  

  #---- get the analysis and assessment weeks ---- 
  analysis_weeks  <- analysis(split)$week_dates 
  assessment_weeks <- assessment(split)$week_dates 
  
  #---- map weeks back to the train data set ----
  analysis_rows <- which(data$week_dates %in% analysis_weeks)
  assessment_rows <- which(data$week_dates %in% assessment_weeks)
  
  #---- turn into a split object ----
  make_splits(
    list(
      analysis = analysis_rows,
      assessment = assessment_rows
      ),
    data
    )
  
}
  
  
  
  
  
#====== Map rolling_origin() splits to train data ======

Weighted Average Percentage Error

Purpose: Define and implement WAPE as the primary evaluation metric to measure relative forecast error scaled by total sales volume, ensuring interpretability across heterogeneous retail demand levels.

##The code below defines a custom weighted absolute percentage error (WAPE) metric in the style recommended by the yardstick package developers. See https://www.tidymodels.org/learn/develop/metrics/ for more information on yardstick developer functions.

#====== Define a WAPE implementation function ======

wape_impl <- function(truth, estimate, case_weights = NULL){
  
  denominator <- sum(abs(truth), na.rm = TRUE) ##make function defensively robust to NA
  
  if (denominator == 0) {
    return(NA_real_)
  }
  
  sum(abs(truth - estimate)) / denominator
}





#====== Define a vector implementation of the WAPE function ======

wape_vec <- function(truth, estimate, na_rm = TRUE, case_weights = NULL, ...) {##new metric functions must have the first 4 arguments to be compatible with the yardstick package
 
  check_numeric_metric(truth, estimate, case_weights) ##verifies that all arguments are numeric

  if (na_rm) {
    
    result <- yardstick_remove_missing(truth, estimate, case_weights) ##removes rows where any argument is missing
    
    ##creates new vectors of clean values
    truth <- result$truth
    estimate <- result$estimate
    case_weights <- result$case_weights
    
    
  } else if (yardstick_any_missing(truth, estimate, case_weights)) { ##returns TRUE if any argument missing
 
    return(NA_real_)
    
  }
  
  wape_impl(truth, estimate, case_weights = case_weights)
}





#====== Define a data frame implementation of the WAPE function ======

wape <- function(data, ...) {
  UseMethod("wape") ##decides which specific function to call based on the class of the object provided
} 


wape.data.frame <- function(data, truth, estimate, na_rm = TRUE, case_weights = NULL, ...) {

  numeric_metric_summarizer(
    name = "wape",
    fn = wape_vec,
    data = data,
    truth = !!enquo(truth), 
    estimate = !!enquo(estimate),
    na_rm = na_rm,
    case_weights = !!enquo(case_weights)
  )
  
}





#====== Add WAPE metric ======

wape <- new_numeric_metric(wape, "minimize")

##wape metric has the same class as yardstick base metrics and can now be used for tuning
